-- создание stats-строки при создании заявки
CREATE OR REPLACE FUNCTION trg_requests_create_stats()
RETURNS trigger AS $$
BEGIN
  INSERT INTO request_offer_stats(request_id)
  VALUES (NEW.id)
  ON CONFLICT (request_id) DO NOTHING;

  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS requests_create_stats ON requests;
CREATE TRIGGER requests_create_stats
AFTER INSERT ON requests
FOR EACH ROW
EXECUTE FUNCTION trg_requests_create_stats();


-- timestamps on insert + closed_at
CREATE OR REPLACE FUNCTION trg_requests_timestamps()
RETURNS trigger AS $$
BEGIN
  IF TG_OP = 'INSERT' THEN
    NEW.created_at := COALESCE(NEW.created_at, now());
    NEW.updated_at := COALESCE(NEW.updated_at, now());
    RETURN NEW;
  END IF;

  -- обновляем updated_at всегда
  NEW.updated_at := now();

  -- если заявка закрывается — фиксируем closed_at
  IF (NEW.status IN ('closed','cancelled')) AND (OLD.status IS DISTINCT FROM NEW.status) THEN
    NEW.closed_at := COALESCE(NEW.closed_at, now());
  END IF;

  -- если заявка переоткрывается — очищаем closed_at
  IF (OLD.status IN ('closed','cancelled'))
     AND (NEW.status IN ('open','review'))
     AND (OLD.status IS DISTINCT FROM NEW.status) THEN
    NEW.closed_at := NULL;
  END IF;

  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS requests_timestamps ON requests;
CREATE TRIGGER requests_timestamps
BEFORE INSERT OR UPDATE ON requests
FOR EACH ROW
EXECUTE FUNCTION trg_requests_timestamps();

-- timestamps on update
CREATE OR REPLACE FUNCTION trg_offers_timestamps()
RETURNS trigger AS $$
BEGIN
  IF TG_OP = 'INSERT' THEN
    NEW.created_at := COALESCE(NEW.created_at, now());
    NEW.updated_at := COALESCE(NEW.updated_at, now());
    RETURN NEW;
  END IF;

-- статистика при создании нового offer
CREATE OR REPLACE FUNCTION trg_offers_stats_on_insert()
RETURNS trigger AS $$
BEGIN
  -- новый offer обязан быть submitted
  IF NEW.status <> 'submitted' THEN
    RAISE EXCEPTION 'New offer must have status submitted';
  END IF;

  -- на всякий случай гарантируем наличие строки в request_offer_stats
  INSERT INTO request_offer_stats(request_id)
  VALUES (NEW.id_request)
  ON CONFLICT (request_id) DO NOTHING;

  -- увеличиваем счётчик неподтверждённых (submitted)
  UPDATE request_offer_stats
  SET count_submitted = count_submitted + 1,
      updated_at = now()
  WHERE request_id = NEW.id_request;

  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS offers_stats_on_insert ON offers;
CREATE TRIGGER offers_stats_on_insert
BEFORE INSERT ON offers
FOR EACH ROW
EXECUTE FUNCTION trg_offers_stats_on_insert();

-- статистика при смене статуса offer
CREATE OR REPLACE FUNCTION trg_offers_stats_on_status_update()
RETURNS trigger AS $$
BEGIN
  IF NEW.status IS DISTINCT FROM OLD.status THEN

    -- submitted -> accepted: count_submitted--, count_accepted_total++
    IF OLD.status = 'submitted' AND NEW.status = 'accepted' THEN
      UPDATE request_offer_stats
      SET count_submitted = GREATEST(count_submitted - 1, 0),
          count_accepted_total = count_accepted_total + 1,
          updated_at = now()
      WHERE request_id = NEW.id_request;
      RETURN NEW;
    END IF;

    -- submitted -> rejected: count_submitted--, count_rejected_total++
    IF OLD.status = 'submitted' AND NEW.status = 'rejected' THEN
      UPDATE request_offer_stats
      SET count_submitted = GREATEST(count_submitted - 1, 0),
          count_rejected_total = count_rejected_total + 1,
          updated_at = now()
      WHERE request_id = NEW.id_request;
      RETURN NEW;
    END IF;

    -- submitted -> deleted : count_submitted--
    IF OLD.status = 'submitted' AND NEW.status = 'deleted' THEN
      UPDATE request_offer_stats
      SET count_submitted = GREATEST(count_submitted - 1, 0),
          updated_at = now()
      WHERE request_id = NEW.id_request;
      RETURN NEW;
    END IF;

    -- accepted -> deleted (уведомление): count_deleted_alert++
    IF OLD.status = 'accepted' AND NEW.status = 'deleted' THEN
      UPDATE request_offer_stats
      SET count_deleted_alert = count_deleted_alert + 1,
          updated_at = now()
      WHERE request_id = NEW.id_request;
      RETURN NEW;
    END IF;

  END IF;

  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS offers_stats_on_status_update ON offers;
CREATE TRIGGER offers_stats_on_status_update
AFTER UPDATE OF status ON offers
FOR EACH ROW
EXECUTE FUNCTION trg_offers_stats_on_status_update();

-- если один offer принят, остальные автоматически отклоняются
CREATE OR REPLACE FUNCTION trg_offers_accept_reject_others()
RETURNS trigger AS $$
BEGIN
  -- срабатываем только при переходе в accepted
  IF OLD.status IS DISTINCT FROM NEW.status
     AND NEW.status = 'accepted' THEN

    UPDATE offers
    SET status = 'rejected'
    WHERE id_request = NEW.id_request
      AND id <> NEW.id
      AND status = 'submitted';
  END IF;

  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS offers_accept_reject_others ON offers;
CREATE TRIGGER offers_accept_reject_others
AFTER UPDATE OF status ON offers
FOR EACH ROW
EXECUTE FUNCTION trg_offers_accept_reject_others();
